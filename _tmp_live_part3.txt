' : l.level === 'success' ? 'log-success' : l.level === 'agent' ? 'log-agent' : 'log-info';
    return `<div class="log-line ${cls}"><span class="log-time">${(l.time||'').slice(11,19)}</span><span class="log-text">${escHtml(l.message)}</span></div>`;
  }).join('');
}

function renderAgents(el) {
  const agents = cachedData.agent_names || [];
  if (!agents.length) { el.innerHTML = '<div class="empty-state">Sem agentes carregados.</div>'; return; }
  const colors = ['#3fb950','#58a6ff','#d29922','#a371f7','#f78166'];
  el.innerHTML = agents.map((a,i) => {
    const color = colors[i % colors.length];
    return `<div class="agent-row"><div class="agent-dot" style="background:${color}"></div><span class="agent-name">${escHtml(String(a.name||a))}</span><span class="agent-role">${escHtml(String(a.role||'agent'))}</span></div>`;
  }).join('');
}

function renderTasks(el) {
  const tasks = cachedData.tasks || [];
  if (!tasks.length) { el.innerHTML = '<div class="empty-state">Sem tarefas no backlog.</div>'; return; }
  el.innerHTML = tasks.slice(0,30).map(t => {
    const s = (t.status||'').toLowerCase();
    const color = s.includes('done')||s.includes('complet') ? '#3fb950' : s.includes('fail')||s.includes('error') ? '#f85149' : '#d29922';
    return `<div class="agent-row"><div class="agent-dot" style="background:${color}"></div><span class="agent-name" style="font-size:12px">${escHtml(t.title||t.desc||t.id||'...')}</span><span class="agent-role">${escHtml(t.status||'?')}</span></div>`;
  }).join('');
}

// ...... Data loading ..........................................................................................................................................................................................
async function loadData() {
  try {
    const resp = await fetch('/api/ecosystem', {signal: AbortSignal.timeout(8000)});
    if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
    const d = await resp.json();
    cachedData = d;

    // KPIs
    document.getElementById('kAgents').textContent = d.agents ?? '...';
    document.getElementById('kAgentsSub').textContent = `${d.agent_names?.length||0} carregados`;
    document.getElementById('kDone').textContent = d.tasks_done ?? '...';
    document.getElementById('kPending').textContent = d.tasks_pending ?? '...';
    document.getElementById('kFailed').textContent = d.tasks_failed ?? '...';
    document.getElementById('lastUpdate').textContent = 'Actualizado ' + new Date().toLocaleTimeString('pt');

    renderTab();
  } catch(e) {
    document.getElementById('lastUpdate').textContent = 'Erro: ' + e.message;
  }
}
