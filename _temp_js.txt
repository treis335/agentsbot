
const API = '';  // mesmo host — usa proxy do dashboard
let currentTab = 'logs';
let cachedData = {};

// ── Tab switching ─────────────────────────────────────────────────────────────
function switchTab(tab, btn) {
  currentTab = tab;
  document.querySelectorAll('.tab').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  const titles = {logs:'Logs do Sistema', agents:'Agentes Ativos', tasks:'Tarefas'};
  document.getElementById('tabTitle').textContent = titles[tab];
  renderTab();
}

function renderTab() {
  const el = document.getElementById('mainContent');
  if (currentTab === 'logs') renderLogs(el);
  else if (currentTab === 'agents') renderAgents(el);
  else renderTasks(el);
}

function renderLogs(el) {
  const logs = cachedData.logs || [];
  if (!logs.length) { el.innerHTML = '<div class="empty-state">Sem logs. Inicia o main.py para ver actividade.</div>'; return; }
  el.innerHTML = [...logs].reverse().map(l => {
    const cls = l.level === 'error' ? 'log-error' : l.level === 'warning' ? 'log-warning' : l.level === 'success' ? 'log-success' : l.level === 'agent' ? 'log-agent' : 'log-info';
    return `<div class="log-line ${cls}"><span class="log-time">${(l.time||'').slice(11,19)}</span><span class="log-text">${escHtml(l.message)}</span></div>`;
  }).join('');
}

function renderAgents(el) {
  const agents = cachedData.agent_names || [];
  if (!agents.length) { el.innerHTML = '<div class="empty-state">Sem agentes carregados.</div>'; return; }
  const colors = ['#3fb950','#58a6ff','#d29922','#a371f7','#f78166'];
  el.innerHTML = agents.map((a,i) => {
    const color = colors[i % colors.length];
    return `<div class="agent-row"><div class="agent-dot" style="background:${color}"></div><span class="agent-name">${escHtml(String(a.name||a))}</span><span class="agent-role">${escHtml(String(a.role||'agent'))}</span></div>`;
  }).join('');
}

function renderTasks(el) {
  const tasks = cachedData.tasks || [];
  if (!tasks.length) { el.innerHTML = '<div class="empty-state">Sem tarefas no backlog.</div>'; return; }
  el.innerHTML = tasks.slice(0,30).map(t => {
    const s = (t.status||'').toLowerCase();
    const color = s.includes('done')||s.includes('complet') ? '#3fb950' : s.includes('fail')||s.includes('error') ? '#f85149' : '#d29922';
    return `<div class="agent-row"><div class="agent-dot" style="background:${color}"></div><span class="agent-name" style="font-size:12px">${escHtml(t.title||t.desc||t.id||'—')}</span><span class="agent-role">${escHtml(t.status||'?')}</span></div>`;
  }).join('');
}

// ── Data loading ──────────────────────────────────────────────────────────────
async function loadData() {
  try {
    const resp = await fetch('/api/ecosystem/status', {signal: AbortSignal.timeout(8000)});
    if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
    const d = await resp.json();
    cachedData = d;

    // KPIs
    document.getElementById('kAgents').textContent = d.agents ?? '—';
    document.getElementById('kAgentsSub').textContent = `${d.agent_names?.length||0} carregados`;
    document.getElementById('kDone').textContent = d.tasks_done ?? '—';
    document.getElementById('kPending').textContent = d.tasks_pending ?? '—';
    document.getElementById('kFailed').textContent = d.tasks_failed ?? '—';
    document.getElementById('lastUpdate').textContent = 'Actualizado ' + new Date().toLocaleTimeString('pt');

    renderTab();
  } catch(e) {
    document.getElementById('lastUpdate').textContent = 'Erro: ' + e.message;
  }
}

// ── Chat ─────────────────────────────────────────────────────────────────────
function escHtml(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
}

function timeNow() {
  return new Date().toLocaleTimeString('pt', {hour:'2-digit',minute:'2-digit'});
}

function appendMsg(role, text) {
  const wrap = document.getElementById('chatMessages');
  const div = document.createElement('div');
  div.className = `msg msg-${role}`;
  if (role === 'agent') div.innerHTML = `<strong>Supervisor</strong>${escHtml(text)}<div class="msg-time">${timeNow()}</div>`;
  else div.innerHTML = `${escHtml(text)}<div class="msg-time">${timeNow()}</div>`;
  wrap.appendChild(div);
  wrap.scrollTop = wrap.scrollHeight;
  return div;
}

function showTyping() {
  const wrap = document.getElementById('chatMessages');
  const div = document.createElement('div');
  div.className = 'typing';
  div.id = 'typingIndicator';
  div.innerHTML = '<span></span><span></span><span></span>';
  wrap.appendChild(div);
  wrap.scrollTop = wrap.scrollHeight;
}

function hideTyping() {
  const el = document.getElementById('typingIndicator');
  if (el) el.remove();
}

async function sendMessage() {
  const input = document.getElementById('chatInput');
  const btn = document.getElementById('sendBtn');
  const msg = input.value.trim();
  if (!msg) return;

  input.value = '';
  input.style.height = '38px';
  btn.disabled = true;
  appendMsg('user', msg);
  showTyping();

  try {
    const resp = await fetch('/api/chat', {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify({message: msg}),
      signal: AbortSignal.timeout(60000),
    });
    const data = await resp.json();
    hideTyping();
    appendMsg('agent', data.response || data.error || 'Sem resposta');
  } catch(e) {
    hideTyping();
    appendMsg('agent', `Erro ao contactar o agente: ${e.message}. Verifica se o main.py está a correr.`);
  } finally {
    btn.disabled = false;
    input.focus();
  }
}

function handleKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendMessage();
  }
}

// ── Init ─────────────────────────────────────────────────────────────────────
loadData();
setInterval(loadData, 5000);
